│   └── analyse_item_list_Coppel Mx (8).csv                      (~79 KB)

│           • Información de shipping, tipo de refrigerador, etc.
│
├── 🐍 SCRIPTS DE ANÁLISIS
│   ├── quick_analysis.py                   # Script de análisis rápido (ejecutable)
│   │   └── Genera reporte ejecutivo en terminal
│   │       ✓ Métricas generales
│   │       ✓ Análisis por marca
│   │       ✓ Recomendaciones estratégicas
│   │
│   └── start_jupyter.sh                    # Script para iniciar Jupyter (ejecutable)
│       └── Inicia ambiente conda y Jupyter Notebook
│
├── 📓 NOTEBOOKS
│   └── analysis_notebook.ipynb             # Notebook principal de análisis (~34 KB)
│       └── 7 secciones de análisis completo:
│           1. Configuración y Carga de Datos
│           2. Limpieza y Validación
│           3. Análisis Exploratorio (EDA)
│           4. Análisis de Competitividad
│           5. Análisis de Atributos Técnicos
│           6. Visualizaciones Avanzadas (Plotly)
│           7. Insights y Recomendaciones
│
└── 📁 outputs/                             # Directorio de salida (auto-generado)
    ├── productos_sin_competencia.csv       # Productos sin match
    ├── productos_fuera_stock.csv           # Productos no disponibles
    ├── top_descuentos.csv                  # Top productos con descuento
    └── estadisticas_por_marca.csv          # Métricas agregadas por marca
